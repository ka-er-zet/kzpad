  <script>
    document.addEventListener('DOMContentLoaded', () => {
      const radios = document.querySelectorAll('input[type="radio"]');

      function updateUI() {
        const groups = [...new Set(Array.from(radios).map(r => r.name))];
        groups.forEach(name => {
          const selected = document.querySelector(`input[name="${name}"]:checked`);
          const parent = document.querySelector(`input[name="${name}"]`).closest('.checklist-item');
          
          if (parent) {
            parent.classList.remove('is-pass', 'is-fail', 'is-na');
            if (selected) {
              parent.classList.add(`is-${selected.value}`);
            }
          }
        });
      }

      radios.forEach(r => r.addEventListener('change', updateUI));
    });
  </script>
